{{- /* Returns the page title with the series part number appended when one can be recovered from the original source URL (e.g. ".../art_and_pets5.html" -> "Art and Pets 5"). Part 1 is conventionally unnumbered upstream, so only parts >= 2 are appended. Display-time only: does not affect .Title, the slug, or the URL. */ -}} {{- $title := .Title -}} {{- with .Params.original_url -}} {{- $m := findRESubmatch `[a-zA-Z_]([0-9]+)\.html?$` . 1 -}} {{- with $m -}} {{- $digits := strings.TrimLeft "0" (index (index . 0) 1) -}} {{- with $digits -}} {{- $n := int . -}} {{- if ge $n 2 -}} {{- $title = printf "%s %d" $title $n -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- $title -}}
